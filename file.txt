          $('#mmg_regist').animate({height:0,marginTop:0,marginBottom:0,opacity:0},300);
          $('#mmg_regist_complete').delay(100).fadeIn(300).delay(6000).fadeOut(1000);
        }else if(response.result == "registed"){
          $('#mmg_regist').animate({height:0,marginTop:0,marginBottom:0,opacity:0},300);
          $('#mmg_registed').delay(100).fadeIn(300);
        }else{
          $("#mmg_regist").find(".error_area").text(response.messages);
        }
      }
    });
  });
  setTextDesc();
  $('input[type="submit"]').click(function(){
    removeSubmitDesc();
  });
});
function setTextDesc(){
  $('#mmg_user_mail').each(function(){initTextboxWithDesc($(this), 'メールアドレスを入力');});
}
function initTextboxWithDesc(input_obj, text){
  setDesc(input_obj, text);
  input_obj.focus(function(e){removeDesc(input_obj);});
  input_obj.blur(function(e){setDesc(input_obj, text);});
}
function removeDesc(input_obj){
  if(input_obj.data('desc')==true){
    input_obj.removeClass('description');
    input_obj.data('desc', null);
    input_obj.val('');
  }
}
function setDesc(input_obj, text){
  if(input_obj.val().length==0||input_obj.val()==text){
    input_obj.addClass('description');
    input_obj.data('desc', true);
    input_obj.val(text);
  }
}
function removeSubmitDesc(){
  if($('#mmg_user_mail').val() == 'メールアドレスを入力'){ $('#mmg_user_mail').val(''); }
}
